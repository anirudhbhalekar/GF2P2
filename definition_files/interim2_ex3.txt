% Varied inputs test %
DEFINE
    clk1 AS CLOCK WITH cycle_rep = 10,
    sw1 AS SWITCH WITH initial = 1,
    sw2 AS SWITCH WITH initial = 0,
    gate1 AS NAND WITH input = 2,
    gate2 AS OR WITH input = 2,
    dtype1 AS DTYPE;

CONNECT
    gate1.I0 = sw1.DATA,
    gate1.I1 = sw2.DATA,
    gate2.I0 = gate1.Q,
    gate2.I1 = clk1.CYCLE,
    dtype1.CLEAR = sw2.DATA,
    dtype1.CLOCK = clk1.CYCLE,
    dtype1.SET = gate2.Q;

MONITOR
    gate1.Q,
    gate2.Q,
    dtype1.Q,
    dtype1.QBAR;

END;
