DEFINE 
    CLK1 AS CLOCK WITH cyclerep=1000, % ERROR INCORRECT PARAM 
    SW1 AS SWITCH WITH initial=1,  

    G1 AS AND WITH inputs=2, 
    G2 AS XOR,
    DTYPE1 AS DTYPE,
    DTYPE2 AS S_TYPE, % ERROR UNDEFINED DEVICE
    DTYPE3 AS D_TYPE; % ERROR UNDEFINED DEVICE

CONNECT CLK1 = DTYPE1.CLK,
    DTYPE3.CLK = CLK1,
    G1.I1 = CLK1,
    DTYPE1.DATA = DTYPE1.QBAR,
    DTYPE2.CLK = DTYPE1.Q,
    DTYPE2.DATA = DTYPE2.QBAR,
    DTYPE3.DATA = DTYPE2.Q,
    G1.I2 = DTYPE3.Q,
    G2.I1 = G1,
    G2.I2 = SW1;

MONITOR 
    DTYPE1.Q, 
    DTYPE2.Q, 
    G2;

END;
