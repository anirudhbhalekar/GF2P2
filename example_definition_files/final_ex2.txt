DEFINE 
	D1 AS DTYPE, 
	D2 AS DTYPE, 
	A1 AS AND WITH inputs = 2, 
	A2 AS AND WITH inputs = 2, 
	N3 AS NAND WITH inputs = 2,
	N4 AS NAND WITH inputs = 2, 
	O1 AS OR WITH inputs = 2, 
	O2 AS OR WITH inputs = 2, 
	S1 AS SWITCH WITH initial = 0, 
	CLK1 AS CLOCK WITH cycle_rep = 5; 


CONNECT
	D1.CLK = CLK1, 
	D1.DATA = O1, 
	D1.SET = S1, 
	D1.CLEAR = S1, 
	A1.I1 = D1.Q,
	A1.I2 = D2.QBAR,
	A2.I1 = D1.QBAR, 
	A2.I2 = D2.Q, 
	O1.I1 = A1, 
	O1.I2 = A2, 
	D2.SET = S1, 
	D2.CLEAR = S1, 
	D2.CLK = CLK1, 
	D2.DATA = O2, 
	N3.I1 = D1.Q, 
	N3.I2 = D2.QBAR, 
	N4.I1 = D2.QBAR, 
	N4.I2 = D1.QBAR, 
	O2.I1 = N3,
	O2.I2 = N4;

MONITOR
	D1.Q, 
	D2.Q; 

END;
